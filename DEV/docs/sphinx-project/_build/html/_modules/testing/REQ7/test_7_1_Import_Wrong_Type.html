def login(driver):
    '''Function responsible for login in the application with a valid account
    
    Arguments:
        driver (webdriver): Selenium webdriver created in the main

    Returns:
        None
    
    Raises:
        If the page loaded have a title difrent from "Login Page"
    '''
    driver.get("http://localhost:3000/login")

    if not "Login Page" in driver.title:
        raise Exception ("Unable to load Login Page")

    driver.maximize_window()

    username = "justAUser1"
    password = "admin123"

    WebDriverWait(driver, 5).until(EC.presence_of_element_located((By.ID, "fullName"))) # wait for the page to load
    driver.find_element(By.ID, "fullName").send_keys(username) # send username and password and click login
    driver.find_element(By.ID, "pass").send_keys(password)
    login_button = driver.find_element(By.ID, "login")
    login_button.click()

    return

@allure.severity(allure.severity_level.NORMAL)
@allure.feature("REQ7 - Pagina de Admin")
@allure.story("Import - Wrong Type")
def test_import_wrong_type():
    '''Function responsible for testing the import of a file of a wrong type

    Arguments:
        driver (webdriver): Selenium webdriver created in the main

    Returns:
        1 if the test was a sucess and 0 if it failed
    '''
    i = 0

    driver = webdriver.Chrome()
    login(driver)

    WebDriverWait(driver, 5).until(EC.presence_of_element_located((By.XPATH, "//*[@id=\"links_nav\"]/li[1]/a/b"))) # wait for the page to load

    admin_button = driver.find_element(By.XPATH, "//*[@id=\"links_nav\"]/li[1]/a/b") # goes to the admin page

    admin_button.click()

    PATH = os.getcwd()
    #TODO: REVIEW IF ELSE STATEMENT
    if PATH.endswith("quirked-up-software"):
        PATH += "\\DEV\\testing\\REQ7"
        os.chdir(PATH)

    elif PATH.endswith("DEV"):
        PATH += "\\testing\\REQ7"
        os.chdir(PATH)
    
    driver.find_element(By.XPATH,"//*[@id=\"root\"]/div/div/div[1]/input").send_keys(os.getcwd()+"//scary_type_import_test.pdf") # clicks on "browse..." an <input type=file> that will receive the file and send the file

    import_button = driver.find_element(By.XPATH, "//*[@id=\"root\"]/div/div/div[1]/button") # clicks on import button
    import_button.click()

    alert = WebDriverWait(driver, 5).until(EC.alert_is_present()) # waits for the alert to pop up 
    msg = alert.text # gets the alert message

    if msg.startswith("not well-formed"): # checks if the alert message is the expected
        i+=1

    alert.accept()
    
    driver.close()

    if i == 1:
        assert True, "Test has sucess"
    else:
        assert False, "Test has failed"
